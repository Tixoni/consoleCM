# Стартовый скрипт для тестирования эмулятора терминала
# Тестирование базовых команд

echo "=== Тестирование команды pwd ==="
pwd

echo "=== Тестирование команды ls ==="
ls

echo "=== Тестирование команды cd ==="
cd home
pwd
ls

echo "=== Тестирование команды cd .. ==="
cd ..
pwd

echo "=== Тестирование команды cd с абсолютным путем ==="
cd /home/user
pwd

echo "=== Тестирование команды vfs-info ==="
vfs-info

echo "=== Тестирование команды cat ==="
cat example.txt

echo "=== Тестирование команды tac ==="
tac example.txt

echo "=== Тестирование команды rev ==="
rev example.txt

echo "=== Тестирование обработки ошибок ==="

echo "--- Ошибка: несуществующая команда ---"
unknown_command

echo "--- Ошибка: cat без аргументов ---"
cat

echo "--- Ошибка: cat с несуществующим файлом ---"
cat nonexistent.txt

echo "--- Ошибка: cd в файл ---"
cd example.txt

echo "--- Ошибка: cd с несуществующим путем ---"
cd /invalid/path

echo "=== Тестирование переменных окружения ==="
echo "Текущий пользователь: $USER"
echo "Домашняя директория: ${HOME}"

echo "=== Тестирование завершено ==="
