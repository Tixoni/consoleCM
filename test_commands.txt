# Тестирование VFS эмулятора
echo === Начало тестирования VFS ===

# Основные команды
help
pwd
ls
ls C:\

# Тестирование VFS
vfs-info
vfs-ls
vfs-ls /home
vfs-ls /home/user
vfs-ls /system

# Тестирование ошибок
vfs-ls /nonexistent
vfs-info invalid_arg

# Команды реальной ОС
cd %TEMP%
ls
cd

echo === Тестирование завершено ===
